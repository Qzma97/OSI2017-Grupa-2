int countf1(FILE *f)
{
	char junk[1024];
	int count = 0, n = 6, check=1;
	while (n--)
		fgets(junk, sizeof(junk), f);
	while (check)
	{
		fgets(junk, sizeof(junk), f);
		if (strcmp(junk, "---------------------------------------\n") == 0)
			check = 0;
		if(check)
			count++;
	}
	fclose(f);
	return count;
}
